function updateTime() {
const now = new Date();
const pstOptions = { timeZone: 'America/Los_Angeles' };
// Get PST time components
const pstTime = now.toLocaleString('en-US', { ...pstOptions, hour: '2-digit', minute: '2-digit', hour12: false });
const [hours, minutes] = pstTime.split(':');
document.getElementById('currentTime').textContent = `${hours}:${minutes}`;
// Get PST date components
const pstDate = new Date(now.toLocaleString('en-US', pstOptions));
const days = ['Sunday', 'Monday', 'Tuesday', 'Wednesday', 'Thursday', 'Friday', 'Saturday'];
const day = days[pstDate.getDay()];
const year = pstDate.getFullYear();
const month = String(pstDate.getMonth() + 1).padStart(2, '0');
const date = String(pstDate.getDate()).padStart(2, '0');
document.getElementById('currentDate').textContent = `${day} ${year}/${month}/${date}`;
// Get PST hour for mode
const hour = parseInt(hours);
const modeEl = document.getElementById('timeMode');
const iconEl = document.getElementById('modeIcon');
if (hour >= 6 && hour < 12) {
modeEl.textContent = 'Morning';
iconEl.textContent = '';
} else if (hour >= 12 && hour < 18) {
modeEl.textContent = 'Afternoon';
iconEl.textContent = '';
} else if (hour >= 18 && hour < 21) {
modeEl.textContent = 'Evening';
iconEl.textContent = '';
} else {
modeEl.textContent = 'Night';
iconEl.textContent = '';
}
}

// Simulated window sensor (independent from gauge)
function simulateWindowSunlight() {
const hour = new Date().getHours();
const popup = document.getElementById('sunlightPopup');
// Sunlight only possible during daytime (7 AM - 6 PM)
const isDaytime = hour >= 7 && hour <= 18;
if (isDaytime && !sunlightDetected) {
// Random chance of sunlight detection
const peakHours = hour >= 10 && hour <= 15;
const chance = peakHours ? 0.4 : 0.7;
if (Math.random() > chance) {
sunlightDetected = true;
popup.classList.add('show');
// Auto-hide after 5-8 seconds
setTimeout(() => {
sunlightDetected = false;
popup.classList.remove('show');
}, 5000 + Math.random() * 3000);
}
}
}

function loadStats() {
const saved = localStorage.getItem('lightStats');
const todayPST = getPSTDate();
if (saved) {
const data = JSON.parse(saved);
const savedDatePST = data.datePST || '';
// Only load if it's the same day (PST)
if (savedDatePST === todayPST) {
dailySeconds = data.daily || 0;
weeklySeconds = data.weekly || 0;
monthlySeconds = data.monthly || 0;
} else {
// Different day - start fresh, old data is in MongoDB
dailySeconds = 0;
weeklySeconds = 0;
monthlySeconds = 0;
localStorage.removeItem('lightStats');
}
}
updateStatsDisplay();
updateDonutChart();
}

// Helper function to get current date in PST
function getPSTDate() {
const now = new Date();
const pstString = now.toLocaleDateString('en-CA', { timeZone: 'America/Los_Angeles' });
return pstString; // Returns YYYY-MM-DD format
}

async function saveStats() {

// Save to localStorage for statistics (with PST date for day comparison)

date: new Date().toISOString(),
datePST: today,
daily: dailySeconds,
weekly: weeklySeconds,
monthly: monthlySeconds
}));

try {
await fetch('/api/usage/save', {
method: 'POST',
headers: { 'Content-Type': 'application/json' },
body: JSON.stringify({
date: today,
onSeconds: dailySeconds
})
});
} catch (error) {
console.error('Failed to save to MongoDB:', error);
}
}
